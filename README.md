- gopls
- xsel

- yarn
- npm
- clangd

- rust-analyzer

```bash
pacman -S nodejs gopls nodejs gopls xsel yarn npm rust rust-analyzer clangd ranger
python3 -m pip install --user --upgrade pynvim
yay -S lua-language-server
yay -S jdtls
```
